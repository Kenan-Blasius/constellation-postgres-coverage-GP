37 def execute(self, method, endpoint, data=None, token=None): 

38 """ 

39 Execute an HTTP request to the FastAPI application. 

40 """ 

41 headers = {} 

42 if token: 

43 headers['Authorization'] = f'Bearer {token}' 

44 

45 full_url = f"{self.url}/{endpoint}" 

46 try: 

47 if method == 'GET': 

48 response = requests.get(full_url, headers=headers) 

49 elif method == 'POST': 

50 response = requests.post(full_url, data=data, headers=headers) 

51 elif method == 'PUT': 

52 response = requests.put(full_url, data=data, headers=headers) 

53 elif method == 'DELETE': 

54 response = requests.delete(full_url, headers=headers) 

55 elif method == 'PATCH': 

56 response = requests.patch(full_url, data=data, headers=headers) 

57 else: 

58 raise ValueError(f"Unsupported method {method} for {full_url}") 

59 

60 return response 

61 except Exception as e: 

62 print(f"Error occurred: {e}") 

63 raise
